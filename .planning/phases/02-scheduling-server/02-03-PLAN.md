  <files>internal/cli/daemon.go</files>
  <action>
    Update `daemonRunCmd.RunE` in `internal/cli/daemon.go` with the full daemon lifecycle wiring:
    
    1. **Config loading**: Load config, set daemon port from flag or config
    2. **DaemonAdapter**: Create adapter types that implement `api.BackupRunner` and `api.JobManager` interfaces, bridging scheduler and config to the API:
       
       ```go
       // daemonRunnerAdapter wraps scheduler + runner for API consumption
       type daemonRunnerAdapter struct {
           scheduler *scheduler.Scheduler
       }
       func (a *daemonRunnerAdapter) RunJob(jobID string) { /* trigger async */ }
       func (a *daemonRunnerAdapter) GetScheduledJobs() []string { return a.scheduler.GetScheduledJobs() }
       func (a *daemonRunnerAdapter) IsRunning() bool { return a.scheduler.IsRunning() }
       func (a *daemonRunnerAdapter) Start() { a.scheduler.Start() }
       func (a *daemonRunnerAdapter) Stop() { a.scheduler.Stop() }
       
       type daemonJobManagerAdapter struct {
           cfg *config.Config
       }
       func (a *daemonJobManagerAdapter) GetJob(jobID string) *models.BackupJob { return a.cfg.GetJob(jobID) }
       func (a *daemonJobManagerAdapter) GetAllJobs() []models.BackupJob { return a.cfg.Jobs }
       ```
    
    3. **Orchestration flow**:
       ```go
       // 1. Create scheduler
       sched := scheduler.NewScheduler(cfg.Daemon.SchedulerInterval, log.Default())
       
       // 2. Create backup job runner (calls the actual backup pipeline)
       runner := scheduler.NewBackupJobRunner(cfg, cfg.GetJob, runBackupForDaemon)
       
       // 3. Create schedule manager and load schedules
       scheduleMgr := scheduler.NewScheduleManager(cfg, sched, runner)
       scheduleMgr.LoadAllSchedules()
       
       // 4. Start scheduler
       sched.Start()
       log.Printf("Scheduler started (interval: %v)", cfg.Daemon.SchedulerInterval)
       
       // 5. Create API adapters
       runnerAdapter := &daemonRunnerAdapter{scheduler: sched}
       jobAdapter := &daemonJobManagerAdapter{cfg: cfg}
       
       // 6. Start API server
       if !daemonNoAPI {
           apiServer := api.NewAPIServer(cfg.Daemon.APIPort, runnerAdapter, jobAdapter)
           if err := apiServer.Start(); err != nil {
               return fmt.Errorf("failed to start API server: %w", err)
           }
           log.Printf("API server listening on 127.0.0.1:%d", cfg.Daemon.APIPort)
       }
       
       // 7. Start system tray
       if !daemonNoTray {
           tray := tray.NewTrayApp(cfg.Daemon.APIPort)
           go tray.Run()
           log.Println("System tray started")
       }
       
       // 8. Signal handling
       sigChan := make(chan os.Signal, 1)
       signal.Notify(sigChan, syscall.SIGINT, syscall.SIGTERM)
       <-sigChan
       
       // 9. Graceful shutdown
       log.Println("Shutting down daemon...")
       // Stop in reverse order: tray, API, scheduler
       if tray != nil { tray.Stop() }
       if apiServer != nil { apiServer.Stop() }
       sched.Stop()
       log.Println("Daemon stopped")
       ```
    
    4. **Store daemon PID**: Write PID file to `~/.ds3backup/daemon.pid` on start, remove on stop. This enables `daemon status` and `daemon stop` to work without the API running (by checking if the PID file exists and the process is alive).
    
    Use existing atomic file write pattern from `internal/config/config.go` for the PID file.
